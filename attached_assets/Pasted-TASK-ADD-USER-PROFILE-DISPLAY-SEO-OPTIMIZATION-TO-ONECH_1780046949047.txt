🔧 STEP 2 — SEND FULL PROFILE ON MATCH

Update match function:

function tryMatch() {
  if (queue.length < 2) return;

  const user1 = queue.shift();
  const user2 = queue.shift();

  user1.partner = user2.id;
  user2.partner = user1.id;

  user1.emit("match_found", {
    partner: user2.userData
  });

  user2.emit("match_found", {
    partner: user1.userData
  });
}
